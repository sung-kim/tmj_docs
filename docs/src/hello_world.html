// vim: expandtab tabstop=3 softtabstop=3 shiftwidth=3
//
// Minimal skeleton for constructing new software tests for tm_junior.
// A test written this way can be run with either:
//
//    `make TESTNAME=<name>.c RUN_RELEASE=1` [extra flags]
//
// Or, if compiling RTL from source:
//
//    `make TESTNAME=<name>.c BUILD_RELEASE=1` [extra flags]

#include "tmj_rt_basic.h"
#include "test_util.h"

int main()
{
   main_sync();
   RT_INIT(NUM_THREADS, NUM_TILES, CTRL_SP_SCOPE_SHRD, r2r_switch_e::DISABLE);

   test_setup();
   PRINT("Hello world");
   test_cleanup();
}
